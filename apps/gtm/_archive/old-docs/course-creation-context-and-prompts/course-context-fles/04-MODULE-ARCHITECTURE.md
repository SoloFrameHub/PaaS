# Module Architecture - Integrated Platform Design

## Overview

The Solo Founder Platform is architected as **three independent but interconnected academies** that share common infrastructure while maintaining clear boundaries. This modular approach enables:

1. **Flexible licensing**: Customers purchase only the modules they need
2. **Independent development**: Each academy can be built and updated separately
3. **Cross-module intelligence**: Shared AI knowledge benefits all modules
4. **Scalable pricing**: From $29/month (single module) to $129/month (complete suite)

## Three Core Modules

```typescript
export function getCheapestPlanForModules(modules: ModuleName[]): PlanName {
  const planEntries = Object.entries(PLANS) as [PlanName, PlanConfig][]
  const matchingPlans = planEntries

    .sort((a, b) => a[1].price - b[1].price)
  
  return matchingPlans[0]?.[0] || 'complete'
}
```

## Feature Flag System

### useFeatureAccess Hook

```typescript
// hooks/useFeatureAccess.ts
import { useAuth } from './useAuth'
import { MODULES, ModuleName } from '@/config/modules'

export function useFeatureAccess() {
  const { userData, loading } = useAuth()

  function hasModuleAccess(module: ModuleName): boolean {
    if (!userData) return false
    return userData.modules?.includes(module) || false
  }

  function hasFeature(feature: string): boolean {
    if (!userData) return false
    return userData.modules?.some(module => 
      MODULES[module].features.includes(feature)
    ) || false
  }

  function getAccessibleModules(): ModuleName[] {
    return userData?.modules || []
  }

  function canUpgrade(): boolean {
    if (!userData) return false
    const currentPlan = userData.plan
    return currentPlan !== 'complete'
  }

  return {
    hasModuleAccess,
    hasFeature,
    getAccessibleModules,
    canUpgrade,
    loading,
  }
}
```

### FeatureGate Component

```typescript
// components/common/FeatureGate.tsx
import { useFeatureAccess } from '@/hooks/useFeatureAccess'
import { UpgradePrompt } from './UpgradePrompt'
import { ModuleName } from '@/config/modules'

interface FeatureGateProps {
  module: ModuleName
  children: React.ReactNode
  fallback?: React.ReactNode
}

export function FeatureGate({ module, children, fallback }: FeatureGateProps) {
  const { hasModuleAccess, loading } = useFeatureAccess()

  if (loading) {
    return <div className="animate-pulse">Loading...</div>
  }

  if (!hasModuleAccess(module)) {
    return fallback || <UpgradePrompt requiredModule={module} />
  }

  return <>{children}</>
}
```

```typescript
// components/layout/ModuleSwitcher.tsx
import { useState } from 'react'
import { useRouter } from 'next/navigation'
import { useFeatureAccess } from '@/hooks/useFeatureAccess'
import { MODULES, ModuleName } from '@/config/modules'

export function ModuleSwitcher() {
  const router = useRouter()
  const { getAccessibleModules } = useFeatureAccess()

      </button>
```

## Upgrade Flow

### Upgrade Path Logic

```typescript
// lib/utils/upgradePaths.ts
import { PlanName, PLANS } from '@/config/modules'

export function getUpgradePath(currentPlan: PlanName, desiredModule: string): PlanName {
  const upgradePaths = {
    founder: {
      sales_training: 'growth',      // Jump to growth for both
      sales_enablement: 'growth',
    },
    sales_starter: {
      founder_academy: 'complete',
      sales_enablement: 'growth',
    },
    growth: {
      founder_academy: 'complete',
    },
  }
  
  return upgradePaths[currentPlan]?.[desiredModule] || 'complete'
}

export function calculateProration(currentPlan: PlanName, newPlan: PlanName, daysRemaining: number): number {
  const currentPrice = PLANS[currentPlan].price
  const newPrice = PLANS[newPlan].price
  const priceDiff = newPrice - currentPrice
  
  // Prorate based on days remaining in billing cycle
  const proration = (priceDiff * daysRemaining) / 30
  return Math.max(0, proration)
}
```
