```python
#!/usr/bin/env python3
"""Mark files as READ in the audit ledger."""
import sys, os, re
LEDGER = '/Volumes/ext-data/github/solofame-platform/docs/AUDIT-FULL-2026-04-23.md'
def main():
    if len(sys.argv) < 3:
        print("usage: mark-audited.py <path|-> <clean|findings> [note]", file=sys.stderr); sys.exit(2)
    path_arg, status = sys.argv[1], sys.argv[2]
    note = sys.argv[3] if len(sys.argv) > 3 else ('clean' if status == 'clean' else 'findings filed')
    symbol = {'clean': '✅', 'findings': '🔍'}.get(status)
    if not symbol:
        print(f"bad status {status}", file=sys.stderr); sys.exit(2)
    paths = [l.strip() for l in sys.stdin if l.strip()] if path_arg == '-' else [path_arg]
    paths = [p[2:] if p.startswith('./') else p for p in paths]
    with open(LEDGER) as f: text = f.read()
    updated = 0
    for p in paths:
        pat = re.compile(rf'(\| \d+ \| `{re.escape(p)}` \| )⏳( \| )[^|]*( \|)')
        new_text, n = pat.subn(rf'\g<1>{symbol}\g<2>{note}\g<3>', text)
        if n: text, updated = new_text, updated + n
        else: print(f"NOT FOUND in ledger: {p}", file=sys.stderr)
    with open(LEDGER, 'w') as f: f.write(text)
    total = len(re.findall(r'\| \d+ \|', text))
    pending = len(re.findall(r'\| ⏳ \|', text))
    clean = len(re.findall(r'\| ✅ \|', text))
    findings = len(re.findall(r'\| 🔍 \|', text))
    skipped = len(re.findall(r'\| ⏭️ \|', text))
    print(f"updated {updated} rows. total={total} pending={pending} clean={clean} findings={findings} skipped={skipped}", file=sys.stderr)
if __name__ == '__main__': main()
```
